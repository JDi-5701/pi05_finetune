# pi05_finetune one-line commands.
# Run from ~/ros_ml_ws/pi05_finetune after entering the ros_ml environment.

# enter environment
rosml && cd ~/ros_ml_ws/pi05_finetune

# 1) rewrite raw actions from next observation.state
python fix_raw_action_from_next_state.py --root ./datasets/raw_pick_up_sponge_42 --out ./datasets/raw_pick_up_sponge_42_next_state --overwrite

# 2) convert to 7D axis-angle delta dataset for pi0.5
rm -rf ./datasets/sponge_pi05_aa_delta_42_next_state_v2 && python convert_to_pi05v2.py --root ./datasets/raw_pick_up_sponge_42_next_state --out ./datasets/sponge_pi05_aa_delta_42_next_state_v2 --repo-id sponge_pi05_aa_delta_42_next_state_v2 --rot aa --action-mode delta --chunk-size 50

# 3) train
python train_pi05.py --root ./datasets/sponge_pi05_aa_delta_42_next_state_v2 --repo-id sponge_pi05_aa_delta_42_next_state_v2 --output-dir ./outputs/pi05_aa_delta_42_next_state_v2 --steps 2000 --batch-size 8 --log-every 50 --save-every 500 --num-workers 4 --action-delta-reference chunk-start --input-action-format same-frame-delta

# 4) train with W&B
python train_pi05.py --root ./datasets/sponge_pi05_aa_delta_42_next_state_v2 --repo-id sponge_pi05_aa_delta_42_next_state_v2 --output-dir ./outputs/pi05_aa_delta_42_next_state_v2 --steps 2000 --batch-size 8 --log-every 50 --save-every 500 --num-workers 4 --action-delta-reference chunk-start --input-action-format same-frame-delta --wandb --wandb-project pi05_finetune --wandb-name pi05_aa_delta_42_next_state_v2

# 5) infer/debug final checkpoint
python infer_aa_delta.py --ckpt ./outputs/pi05_aa_delta_42_next_state_v2/final --root ./datasets/sponge_pi05_aa_delta_42_next_state_v2 --repo-id sponge_pi05_aa_delta_42_next_state_v2 --episode-index 0 --chunk-stride 30 --take-steps 30 --rollout-from-chunk-start --plot-out ./outputs/episode0_aa_delta_debug.png --device cuda

# 6) infer/debug a step checkpoint
python infer_aa_delta.py --ckpt ./outputs/pi05_aa_delta_42_next_state_v2/step_2000 --root ./datasets/sponge_pi05_aa_delta_42_next_state_v2 --repo-id sponge_pi05_aa_delta_42_next_state_v2 --episode-index 0 --chunk-stride 30 --take-steps 30 --rollout-from-chunk-start --plot-out ./outputs/episode0_step2000_aa_delta_debug.png --device cuda
